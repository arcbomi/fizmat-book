$$
\begin{align*}
|f(x)| < g(x) & \iff 
\begin{cases} 
f(x) < g(x) \\ 
-f(x) < g(x) 
\end{cases} \\

|f(x)| > g(x) & \iff 
\left[ 
\begin{array}{ll}
f(x) > g(x) \\ 
f(x) < -g(x) 
\end{array} 
\right. \\

f(|x|) < g(x) & \iff 
\left[ 
\begin{array}{ll}
\begin{cases} 
x \geq 0 \\ 
f(x) < g(x) 
\end{cases} \\

\begin{cases} 
x < 0 \\ 
f(-x) < g(x) 
\end{cases}
\end{array} 
\right.
\end{align*}

$$
