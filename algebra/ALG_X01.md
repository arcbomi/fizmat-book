\begin{align*}
\lim_{x\rightarrow c}(f(x)+g(x))&=\lim_{x\rightarrow c}f(x)+\lim_{x\rightarrow c}g(x) \\
\lim_{x\rightarrow c}(f(x)-g(x))&=\lim_{x\rightarrow c}f(x)-\lim_{x\rightarrow c}g(x) \\

\lim_{x\rightarrow c}\frac{f(x)}{g(x)}&=\frac{\lim_{x\rightarrow c}f(x)}{\lim_{x\rightarrow c}g(x)} \\
\end{align*}
